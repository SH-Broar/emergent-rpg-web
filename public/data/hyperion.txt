# ============================================================
# hyperion.txt — 히페리온 시스템 정의
# 각 캐릭터의 히페리온 레벨 조건 (5단계)
# 조건 충족 시 레벨당: HP +10, MP +5, Attack +2, Defense +1, Vigor +5
#
# [__default__] = 탄생 캐릭터용 범용 히페리온 (이름 매칭 없을 때 사용)
# condition_N = 조건 설명문 (UI 표시용)
# condition_N_type = 조건 타입:파라미터
#
# --- 기본 조건 타입 ---
#   manual                       — 수동 확인 (미구현 시스템)
#   days_passed:N                — 게임 N일 경과
#   hyperion_total:N             — 전체 히페리온 레벨 합 N 달성
#   actor_recruited:이름         — 해당 NPC 동료(존재) 확인
#   all_recruited:이름1,이름2    — 모든 NPC 동료 확인
#   recruited_count:N            — 역대 동료 수 N명 이상
#   all_races_recruited          — 모든 종족의 동료 보유
#   all_elements_recruited       — 모든 속성의 동료 보유
#   gold_owned:N                 — 골드 N 이상 보유
#   location_visited:ID          — 해당 지역 방문 기록
#   all_locations_visited        — 모든 장소 방문
#   visited_count:N              — N곳 이상 방문
#   conversation_with:이름       — 해당 NPC와 대화
#   conversation_count:N         — 총 대화 N회 이상
#   dungeon_clear_count:N        — 던전 N회 이상 클리어
#   hyperion_levels:이름1:N1,이름2:N2 — 특정 캐릭터들의 히페리온 레벨 확인
#
# --- 전투·탐험 추적 ---
#   monsters_killed:N            — 몬스터 총 N마리 처치
#   monster_types:N              — 서로 다른 N종 몬스터 사냥
#   damage_dealt:N               — 누적 데미지 N 이상
#   damage_taken:N               — 누적 피격 데미지 N 이상
#   max_damage:N                 — 단일 최대 데미지 N 이상
#   treasure_found:N             — 보물상자 N회 발견
#
# --- 경제·활동 ---
#   gold_spent:N                 — 골드 총 N 이상 소비
#   items_sold:N                 — 아이템 총 N개 판매
#   items_crafted:N              — 아이템 총 N회 제작
#   activities_done:N            — 지역 활동 총 N회 수행
#   food_eaten:N                 — 서로 다른 음식 N종 섭취
#   vigor_spent:N                — 기력 총 N 이상 소비
#
# --- 관계·동료 ---
#   friend_count:N               — 호감도 0.3+ NPC N명 이상
#   companion_days:이름:N        — 이름과 총 N일 동행
#   relationship:이름:F          — 이름과의 호감도 F 이상
#
# --- 스탯·속성 ---
#   stat_total:N                 — 자신의 공격+방어 합 N 이상
#   party_stat_total:N           — 파티 전원 공격+방어 합 N 이상
#   color_value:속성:F           — 특정 속성 값 F 이상
# ============================================================
