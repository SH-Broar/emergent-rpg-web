# Events MVR 단계 이벤트 풀.
# 평균 500~1000자, 단위 완결.

[event.ev-traveler-fork]
name = 갈림길의 여행자
node_kinds = event
weight = 2
once_per_run = true
body = 길이 둘로 갈라진다. 그 가운데에 한 여행자가 쪼그려 앉아 있다. 그는 그대를 보고도 놀라지 않는다. "어느 쪽을 가실 거예요?" 그가 묻는다. "왼쪽은 늪이지만 약초가 있고, 오른쪽은 평평하지만 아무것도 없죠. 저는 어느 쪽도 안 갔어요. 그냥 여기 있는 게 좋아서." 그의 등 뒤에 작은 짐꾸러미가 놓여 있다.

[event.ev-traveler-fork.choice.1]
label = 왼쪽 (약초)
hp = -3
grant_card = c-burn-mark
result_text = 늪을 헤치고 약초를 얻었다. 다리에 작은 상처가 남았다.

[event.ev-traveler-fork.choice.2]
label = 오른쪽 (평탄)
draw = 1
result_text = 평탄한 길로 가서 잠시 생각을 정리했다.

[event.ev-traveler-fork.choice.3]
label = 그와 함께 머문다
heal_pct = 35
gold = -3
result_text = 그의 옆에 앉아 잠시 쉬었다. 그가 빵을 나눠줬다.


[event.ev-stranger-favor]
name = 모르는 이의 부탁
node_kinds = event, village
weight = 1
once_per_run = true
body = 한 노파가 광주리를 떨어뜨리고 안 보이는 듯 그 위를 더듬는다. "이쪽인가… 아니, 그쪽인가…" 그녀의 눈은 흐려 있다. 광주리에서 무언가가 굴러 나오는데, 모양으로 봐선 마법이 깃든 작은 돌이다.

[event.ev-stranger-favor.choice.1]

[event.ev-stranger-favor.choice.2]
label = 돌만 챙긴다
gold = 5
result_text = 노파는 끝내 광주리를 못 찾았다. 그대의 주머니에 돌이 있다.

[event.ev-stranger-favor.choice.3]
label = 그냥 지나친다
result_text = 발걸음이 한참 후에도 가볍지 않았다.

# ============================================================
# 샘물의 약속 condition + custom + followup chain.
# 깊은 컬러를 가진 자에게만 완전한 회복을 약속하는 샘.
# ============================================================

[event.ev-fountain-vow]
name = 약속하는 샘
node_kinds = event, rest
weight = 1
once_per_run = true
body = 길가 돌틈에 작은 샘이 있다. 물 표면에 글자 같은 무늬가 천천히 움직인다. 마시기 전에 약속하라는 뜻인 것 같다. "당신의 색이 충분히 깊다면, 나는 그대의 몸을 완전히 새로 짠다."

[event.ev-fountain-vow.choice.1]
label = 한 모금 마신다 (약속 없이)
hp = -3
result_text = 물 맛이 시다. 글자가 흐려지더니 마지막 줄만 남았다. "다음엔, 약속을 가져오세요."

[event.ev-fountain-vow.choice.2]
label = 약속의 무게를 측정한다
gold = -5
followup = ev-fountain-vow-aftermath
result_text = 동전을 던지자 잔물결이 일었다. 샘이 그대의 색을 다시 본다 잠시 후 무언가가 등 뒤에서 다가온다.

[event.ev-fountain-vow.choice.3]
label = 물 깊은 자의 자격으로 마신다
condition = color:water>=15
custom = heal-full
result_text = 글자가 환하게 빛났다. 물이 갈비뼈 안쪽까지 흘러 들어왔다 몸이 새로 짜였다.

[event.ev-fountain-vow.choice.4]
label = 그대로 떠난다
draw = 1
result_text = 등 뒤에서 샘이 천천히 잔잔해진다. 다음에 다시 와도 같은 약속을 묻을 것이다.

# 샘의 후속 측정을 선택한 자에게만. followup으로 연결.
# ============================================================

[event.ev-fountain-vow-aftermath]
name = 약속의 뒷자리
node_kinds = event
weight = 0
once_per_run = true
body = 샘에서 멀어진 뒤에도 잔물결의 소리가 등 뒤를 따라왔다. 길 끝에 작은 돌무더기가 새로 쌓여 있다. 누군가가 그 위에 유리병 하나를 두고 갔다.

[event.ev-fountain-vow-aftermath.choice.1]
label = 병을 들어 본다
grant_card = c-recover
result_text = 병 안에 빛이 흔들렸다. 손이 무거워졌고 들고 갈 게 하나 늘었다.

[event.ev-fountain-vow-aftermath.choice.2]
label = 돌무더기에 손을 얹는다
custom = pulse-all-colors-3
result_text = 손바닥이 여덟 방향으로 떨렸다. 몸 안의 색들이 한꺼번에 짙어졌다.

[event.ev-fountain-vow-aftermath.choice.3]
label = 그대로 지나친다
heal_pct = 35
result_text = 등 뒤의 잔물결이 마침내 잦아들었다. 약속이 자기 자리로 돌아간 것 같았다.

[event.ev-time-cracker]
name = 모래시계 상인
node_kinds = event, village
weight = 1
once_per_run = true
body = 노점 위에 모래시계가 가지런히 줄지어 있다. 시계마다 모래의 흐름이 다르다. 어떤 건 거꾸로 떨어지고, 어떤 건 낱알씩만 떨어진다. "잠깐씩만 빌려 드려요. 단, 충분한 무게가 있어야 해요."

[event.ev-time-cracker.choice.1]
label = 시계 하나를 빌린다 (충분한 무게로)
condition = gold>=20
gold = -20
custom = time-extend-10
result_text = 모래시계 하나가 멈췄다. 그 멈춤이 등 뒤로 옮겨와 길이 조금 늘어졌다.

[event.ev-time-cracker.choice.2]
label = 모래 낱알 하나만 받는다
gold = -3
custom = gift-time-shards-3
result_text = 모래 낱알이 손바닥에서 시간의 조각으로 굳었다. 작지만 분명한 무게가 느껴졌다.

[event.ev-time-cracker.choice.3]
label = 상인의 이야기만 듣는다
affinity = npc-mono:1
result_text = 상인이 모노의 표정으로 잠시 웃었다. "이 시간은 빌려드린 게 아니라, 원래 그대 것이었어요."

[event.ev-time-cracker.choice.4]
label = 그대로 떠난다
draw = 1
result_text = 노점을 빠져나오자 모래시계 줄이 하나 줄어들어 있었다.
