# NPC 스파링(안전 대련) 사건 — 친한 호전 NPC와의 겨룸.
# 조건: affinity:npc-X>=3 (친밀도 3 이상이면 풀에 등장). spar= 토큰으로 안전 대련 전투 진입.
# 본문 짧게, prose 규칙 엄수. 금지어: 결, 한 박, 박자.

# === 카요 (npc-cayo) — 드래곤 대장장이. 마노니클라 권역. ===
[event.ev-spar-cayo]
name = 카요의 모루 앞
body = 카요가 망치를 내려놓고 너를 본다. 한 수 받아 보겠냐고, 비늘 사이로 웃는다.
weight = 2
once_per_run = false
node_kinds = event
condition = affinity:npc-cayo>=3
featured_npcs = npc-cayo

[event.ev-spar-cayo.choice.1]
label = 겨뤄 본다
spar = npc-spar-cayo

[event.ev-spar-cayo.choice.2]
label = 다음에
result_text = 카요가 고개를 끄덕이고 다시 모루로 돌아간다.
