#   gold:N            골드 +N
#   time-shards:N     시간의 조각 +N
#   color-boost:KEY:N colors[KEY] += N (KEY: fire/water/electric/iron/earth/wind/light/dark)
#   color-all:N       8 컬러 모두 += N
#   grant-card:ID     카드 한 장 컬렉션 추가
#   grant-relic:ID    유물 한 점 추가

#   combat-draw:N            전투 중 카드 N장 드로우
#   combat-block:N           전투 중 방어 +N
#   combat-enemy-status:S:N  적에게 status S(vulnerable/weakness/poison 등) N 스택
#   combat-self-status:S:N   자신에게 status S(strength 등) N 스택
