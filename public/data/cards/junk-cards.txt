# Junk Cards 몬스터가 손패/덱에 주입하는 잡카드 (Stage 2 카드 교란).
# source = junk → 상점/공방/보상 등 모든 풀에서 제외. 전투 종료 시 자동 소멸(런 덱에 안 남음).
# 세 종류는 "별개의 상태"로 동작:
#   상처(wound) : 사용 불가. 손패 한 칸을 죽은 채로 차지. 풀 수 없음.
#   빈 카드(blank): cost 0 + 소멸. 사용(클릭)하면 그냥 사라짐 — 칸을 비울 순 있는 무해한 잡동사니.
#   저주(curse) : 손에 두면 매 턴 시작 2 직접 피해(curse-tick). 1 마나로 사용=소멸시켜 끊을 수 있다.

[card.c-junk-wound]
name = 상처
rank = basic
source = junk
cost = 1
unplayable = true
trigger = manual
effects =
target_mode = self
cast_speed = normal
flavor = 자꾸 욱신거려서 신경이 쓰인다.

[card.c-junk-blank]
name = 빈 카드
rank = basic
source = junk
cost = 0
trigger = manual
effects = exhaust-self
target_mode = self
cast_speed = fast
flavor = 아무것도 적혀 있지 않다.

[card.c-junk-curse]
name = 저주
rank = basic
source = junk
cost = 1
trigger = manual
effects = curse-tick:2, exhaust-self
target_mode = self
cast_speed = normal
flavor = 주머니 속에서 자꾸 꼼지락거린다.
